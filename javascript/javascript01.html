01. 변수 : 데이터 저장

{
    var x = 100;            //변수 x에 숫자 100을 저장함
    var y = 200;            //변수 y에 숫자 200을 저장함
    var z = "javascript";   //변수 z에 문자 "javascript"를 저장함

    document.write("*** 01. 변수 ***")
    document.write(x);
    document.write(y);
    document.write(z);
}

결과보기

100
200
"javascript"

03. 변수 : 데이터 저장 + 데이터 변경 + 데이터 추가

04. 변수 : 변수의 종류 : 지역변수 + 전역변수 + 매개변수

{
    document.write("*** 04. 변수 : 변수의 종류 ***");

    let x = 100; //전역변수
    let y = 200; //전역변수

    function func() {
    let x = 100;         //function 안은 지역변수
    let z = "javascript";    //지역변수
    x = 200;        //지역변수 100->200
    y = 300;        //전역변수가 200->300

    document.write("함수 안");
    document.write(x);
    document.write(y);
    document.write(z);
    }
    func();

    document.write("함수 밖");
    document.write(x);
    document.write(y);
    document.write(z);      //z는 실행안됨
}

결과보기

(함수 안) 100
300
"javascript"
(함수밖) 100
200
실행x

17. 데이터를 저장(키와 값) : 표현 방법7 : 객체 속 함수

{
    const obj = {
        a: 100,
        b: [200, 300],
        c: {x: 400, y: 500},
        d: "javascript",
        e: function(){
            document.write("javascript가 실행되었습니다.");
        },
        f:function(){
            document.write(obj.d + "가 실행되었습니다.
");
        },
        g:function(){
            document.write(this.d + "가 실행되었습니다.
");
        }
    } 
     document.write("*** 17. 객체 ***");
     document.write(obj.a);   //객체
     document.write(obj.b);
     document.write(obj.b[0]);
     document.write(obj.b[1]);
     document.write(obj.c.x);
     document.write(obj.c.y);
     obj.d();
     obj.e();
     obj.f();
}
