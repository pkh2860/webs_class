02. 수학 메서드 : Math.random() : 0과 1사이의 난수 값을 반환 : 반환(숫자)

번호 기본값 메서드 결괏값 
{
    document.querySelector(".sample2_R1").innerHTML = Math.random();
    document.querySelector(".sample2_R1").style.color = "#000";
    document.querySelector(".sample2_B1").addEventListener("click", ()=>{
        document.querySelector(".sample2_R1").innerHTML = Math.random();
    })

    document.querySelector(".sample2_R2").innerHTML = Math.floor(Math.random()*10);
    document.querySelector(".sample2_R2").style.color = "#000";
    document.querySelector(".sample2_B2").addEventListener("click", ()=>{
        document.querySelector(".sample2_R2").innerHTML = Math.floor(Math.random()*10);
    })

    document.querySelector(".sample2_R3").innerHTML = Math.floor(Math.random()*10 + 10);
    document.querySelector(".sample2_R3").style.color = "#000";
    document.querySelector(".sample2_B3").addEventListener("click", ()=>{
        document.querySelector(".sample2_R3").innerHTML = Math.floor(Math.random()*10 + 10);
    })
}
